	\fi}

\xdef\horizontal{\pgfmathresult}
\foreach \x in {1,...,\horizontal} {
	\pgfmathsetmacro{\pos}{\x + 1}
	\node (\x a0) at (\pos,0) {};}


%edges
%horizontal
\xdef\horizontal{3}
\foreach \y in {\size,...,1}{
	\draw (1a\y)--(\horizontal a\y);
	\pgfmathparse{int(\horizontal+2)}
	\xdef\horizontal{\pgfmathresult}
}
\pgfmathparse{int(\horizontal - 2)}
\xdef\horizontal{\pgfmathresult}
\draw (1a0) -- (\horizontal a0);

%vertical
\xdef\horizontal{3}

	\pgfmathparse{int(2)}

		\pgfmathsetmacro{\next}{\y-1}
		\draw (\x a\y)--(\k a\next);
		\pgfmathparse{int(\k+1)}
		\xdef\k{\pgfmathresult}
	}

	\xdef\horizontal{\pgfmathresult}
